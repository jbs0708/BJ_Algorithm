import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        // <StringBuilder>를 이용하여 추후에 한 번의 print로 출력한다
        StringBuilder sb = new StringBuilder();
        // 문제의 입력이 줄로 띄워져 있어서 br.read()가 아닌 br.readLine()을 이용한다
        int n = Integer.parseInt(br.readLine());

        // 절대값 기준으로 1000000이하이므로 범위를 1000000씩 더해서 [ 0 ~ 2000000 ]로 설정
        boolean[] arr = new boolean[2000001];

        for (int i = 0; i < n; i++) {
	// 입력받은 순서의 배열을 true로 둔다
            arr[Integer.parseInt(br.readLine()) + 1000000] = true;
        }

        for (int i = 0; i < arr.length; i++) {
	// 입력받은 순서가 되면 아까 설정했던 범위값을 빼주고 StringBuilder에 추가한 뒤 줄바꿈해준다
            if (arr[i]) {
                sb.append((i - 1000000)).append('\n');
            }
        }
        System.out.println(sb);
    }
}
