import java.util.Scanner;

public class Main{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int ans = 666;

        int i = 1;
        while (i <= n) {
            String str = "" + ans;
            ans++;
            if(str.contains("666")) {
                i++;
            }
        }

        System.out.println(ans - 1);
    }
}
